# ====================== 4. kb/ingesters/mitre_attck_ingester.py ======================
from attackcti import attackcti
import sqlite3
from tqdm import tqdm
import os

def sync_mitre(db_path: str = "db/kali_tools.db"):
    """Sync latest MITRE ATT&CK data into kali_tools.db"""
    conn = sqlite3.connect(db_path)
    c = conn.cursor()
    
    client = attackcti.AttackCTI()
    # Fetch Enterprise ATT&CK (you can add Mobile/ICS later)
    techniques = client.get_techniques()
    
    with Progress() as progress:
        task = progress.add_task("[cyan]Syncing MITRE ATT&CK...", total=len(techniques))
        for tech in techniques:
            # Map to our schema (simplified)
            tool_name = tech.get("name", "unknown")
            attack_phase = tech.get("kill_chain_phases", [{}])[0].get("phase_name", "unknown")
            
            c.execute("""
                INSERT OR REPLACE INTO kali_tools 
                (tool_name, attack_phase, one_line_desc)
                VALUES (?, ?, ?)
            """, (tool_name, attack_phase, tech.get("description", "")[:200]))
            progress.advance(task)
    
    conn.commit()
    conn.close()
    print(f"✅ MITRE ATT&CK synced ({len(techniques)} techniques)")
